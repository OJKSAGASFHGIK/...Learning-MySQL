--
-- foreign key -- Example of use to foreign key
create table t_test_customers( -- this table are the FKs
	id_fk_customer int primary key auto_increment,
    a_first_name varchar(54),
    a_last_name varchar(54)
);

create table t_test_transactions( -- this table are the PKs that cotains the FKs
	id_pk_payment int primary key auto_increment,
    a_amount_paid decimal(5,2),
    id_fk_customer int,
    constraint fk_customer foreign key(id_fk_customer) references t_test_customers(id_fk_customer)

    foreign key(id_fk_customer) references t_test_customers(id_fk_customer)
);
